15  package com.google.common.html;
16  
17  import com.google.common.annotations.Beta;
18  import com.google.common.annotations.GwtCompatible;
19  import com.google.common.escape.Escaper;
20  import com.google.common.escape.Escapers;
21  
22  /**
23   * {@code Escaper} instances suitable for strings to be included in HTML attribute values and
24   * <em>most</em> elements' text contents. When possible, avoid manual escaping by using templating
25   * systems and high-level APIs that provide autoescaping.
26   * One Google-authored templating system available for external use is <a
27   * href="https://developers.google.com/closure/templates/">Closure Templates</a>.
28   *
29   * <p>HTML escaping is particularly tricky: For example, <a href="http://goo.gl/5TgZb">some
30   * elements' text contents must not be HTML escaped</a>. As a result, it is impossible to escape an
31   * HTML document correctly without domain-specific knowledge beyond what {@code HtmlEscapers}
32   * provides. We strongly encourage the use of HTML templating systems.
33   *
34   * @author Sven Mawson
35   * @author David Beaumont
36   * @since 15.0
37   */
38  @Beta
39  @GwtCompatible
40  public final class HtmlEscapers {
41    /**
42     * Returns an {@link Escaper} instance that escapes HTML metacharacters as specified by <a
43     * href="http://www.w3.org/TR/html4/">HTML 4.01</a>. The resulting strings can be used both in
44     * attribute values and in <em>most</em> elements' text contents, provided that the HTML
45     * document's character encoding can encode any non-ASCII code points in the input (as UTF-8 and
46     * other Unicode encodings can).
47     *
48     * <p><b>Note:</b> This escaper only performs minimal escaping to make content structurally
49     * compatible with HTML. Specifically, it does not perform entity replacement (symbolic or
50     * numeric), so it does not replace non-ASCII code points with character references. This escaper
51     * escapes only the following five ASCII characters: {@code '"&<>}.
52     */
53    public static Escaper htmlEscaper() {
54      return HTML_ESCAPER;
55    }
56  
57    // For each xxxEscaper() method, please add links to external reference pages
58    // that are considered authoritative for the behavior of that escaper.
59  
60    private static final Escaper HTML_ESCAPER =
61        Escapers.builder()
62            .addEscape('"', "&quot;")
63            // Note: "&apos;" is not defined in HTML 4.01.
64            .addEscape('\'', "&#39;")
65            .addEscape('&', "&amp;")
66            .addEscape('<', "&lt;")
67            .addEscape('>', "&gt;")
68            .build();
69  
70    private HtmlEscapers() {}
71  }
