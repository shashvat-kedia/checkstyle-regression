165     /**
166      * @throws FinderException
167      * @throws CreateException
168      * @ejb.interface-method
169      */
170     public List assignStudyToMedia(StudyLocal study, List mediaPool,
171             long maxMediaSize, String prefix) throws FinderException,
172             CreateException {
173         if (mediaPool == null) {
174             mediaPool = getCollectingMedia();
175         }
176         Map instanceFiles = new HashMap();
177         InstanceLocal instance;
178         FileLocal file, file2;
179         int avail, avail2;
180         Collection files;
181         long size = 0;
182         for (Iterator iter = study.getInstancesNotOnMedia().iterator(); iter
183                 .hasNext();) {
184             instance = (InstanceLocal) iter.next();
185             files = instance.getFiles();
186             if (files.isEmpty()) {
187                 log.warn("Instance " + instance.getPk() + "("
188                         + instance.getSopIuid() + ") has no files! ignored!");
189                 continue;
190             }
191             Iterator it = files.iterator();
192             for (file = (FileLocal) it.next(); it.hasNext();) {
193                 file2 = (FileLocal) it.next();
194                 avail = file.getFileSystem().getAvailability();
195                 avail2 = file2.getFileSystem().getAvailability();
196                 if (avail2 < avail) {
197                     file = file2;
198                 } else if (file2.getPk().longValue() > file.getPk().longValue()) {
199                     file = file2;
200                 }
201             }
202             size += file.getFileSize();
203             instanceFiles.put(instance, file);
204         }
205         if (size > maxMediaSize) {
206             log.info("Study size (" + size + ") exceed maxMediaSize ("
207                     + maxMediaSize + ")! Split study is necessary!");
208             splitStudy(instanceFiles, mediaPool, maxMediaSize, prefix);
209         } else {
210             log.debug("assign study " + study.getStudyIuid() + " (size=" + size
211                     + ") to media");
212             assignInstancesToMedia(instanceFiles.keySet(), size, getMedia(
213                     mediaPool, maxMediaSize - size, prefix));
214         }
215         return mediaPool;
216     }

218     /**
219      * 
220      * @param instanceFiles
221      *            map with all instances for a media.
222      * @param maxUsed
223      *            The max mediaUsage a media must have to can store all
224      *            instances.
225      * @param mediaPool
226      *            List of COLLECTING media
227      * @param prefix
228      *            Prefix for Fileset ID. (Used if a new media must be created.
229      * @throws CreateException
230      */
231     private void assignInstancesToMedia(Set instances, long size,
232             MediaLocal media) throws CreateException {
233         log.debug("assign instances (" + instances + ") to media " + media);
234         for (Iterator iter = instances.iterator(); iter.hasNext();) {
235             ((InstanceLocal) iter.next()).setMedia(media);
236         }
237         media.setMediaUsage(media.getMediaUsage() + size);
238     }
239 
240     private MediaLocal getMedia(List mediaPool, long maxUsed, String prefix)
241             throws CreateException {
242         maxUsed++;
243         if (mediaPool.size() > 0) {
244             MediaLocal media;
245             for (Iterator iter = mediaPool.iterator(); iter.hasNext();) {
246                 media = (MediaLocal) iter.next();
247                 if (media.getMediaUsage() < maxUsed)
248                     return media;
249             }
250         }
251         log.debug("We need new media! mediaPool:" + mediaPool);
252         MediaLocal media = createMedia(prefix);
253         mediaPool.add(media);
254         return media;
255 
256     }

258     /**
259      * @param instanceFiles
260      * @param maxMediaSize
261      * @param prefix
262      * @throws CreateException
263      */
264     private void splitStudy(Map instanceFiles, List mediaPool,
265             long maxMediaSize, String prefix) throws CreateException {
266         Map.Entry entry;
267         log.info("Split study!");
268         long fileSize, size = 0;
269         Set instances = new HashSet();
270         for (Iterator iter = instanceFiles.entrySet().iterator(); iter
271                 .hasNext();) {
272             entry = (Map.Entry) iter.next();
273             fileSize = ((FileLocal) entry.getValue()).getFileSize();
274             size += fileSize;
275             if (size > maxMediaSize) {
276                 log.debug("assign instances (" + instances.size() + "/"
277                         + instanceFiles.size() + ") to new media!");
278                 assignInstancesToMedia(instances, size, createMedia(prefix));
279                 size = fileSize;
280                 instances.clear();
281             }
282             instances.add(entry.getKey());
283         }
284         if (!instances.isEmpty()) {
285             log.debug("assign remaining instances (" + instances.size() + "/"
286                     + instanceFiles.size() + ") to new media!");
287             assignInstancesToMedia(instances, size, getMedia(mediaPool,
288                     maxMediaSize - size, prefix));
289         }
290     }

293      * 
294      * @ejb.interface-method
295      */
296     public List getCollectingMedia() throws FinderException {
297         List mediaCollection = (List) mediaHome.findByStatus(MediaDTO.OPEN);
298         Comparator comp = new Comparator() {
299             public int compare(Object arg0, Object arg1) {
300                 MediaLocal ml1 = (MediaLocal) arg0;
301                 MediaLocal ml2 = (MediaLocal) arg1;
302                 return (int) (ml2.getMediaUsage() - ml1.getMediaUsage());// more
303                                                                          // usage
304                                                                          // before
305                                                                          // lower
306                                                                          // usage
307                                                                          // !
308             }
309         };
310         Collections.sort(mediaCollection, comp);
311         log.debug("Number of 'COLLECTING' media found:"
312                 + mediaCollection.size());
313         return mediaCollection;
314     }

353     /**
354      * Find media for given search params.
355      * <p>
356      * Add all founded media to the given collection.<br>
357      * This allows to fill a collection with sequential calls without clearing
358      * the collection.<br>
359      * 
360      * @param col
361      *            The collection to store the result.
362      * @param after
363      *            'created after' Timestamp in milliseconds
364      * @param before
365      *            'created before' Timestamp in milliseconds
366      * @param stati
367      *            Media status (<code>null</code> to get all media for given
368      *            time range)
369      * @param offset
370      *            Offset of the find result. (used for paging.
371      * @param limit
372      *            Max. number of results to return. (used for paging)
373      * @param desc
374      *            Sort order. if true descending, false ascending order.
375      * 
376      * @return The total number of search results.
377      * 
378      * @ejb.interface-method
379      */
380     public int findByCreatedTime(Collection col, Long after, Long before,
381             int[] stati, Integer offset, Integer limit, boolean desc)
382             throws FinderException {
383         Timestamp tsAfter = null;
384         if (after != null)
385             tsAfter = new Timestamp(after.longValue());
386         Timestamp tsBefore = null;
387         if (before != null)
388             tsBefore = new Timestamp(before.longValue());
389         col.addAll(toMediaDTOs(mediaHome.listByCreatedTime(stati, tsAfter,
390                 tsBefore, offset, limit, desc)));
391         return mediaHome.countByCreatedTime(stati, tsAfter, tsBefore);
392     }

437      * objects.
438      * 
439      * @param c

441      *

444     private List toMediaDTOs(Collection c) {
445         ArrayList list = new ArrayList();
446         for (Iterator it = c.iterator(); it.hasNext();) {
447             list.add(toMediaDTO((MediaLocal) it.next()));
448         }
449         return list;

452     /**
453      * Creates a MediaDTO object for given given MediaLocal object.
454      * 
455      * @param media
456      *            A MediaLocal object.
457      * 
458      * @return The MediaDTO object for given MediaLocal.
459      */
460     private MediaDTO toMediaDTO(MediaLocal media) {
461         MediaDTO dto = new MediaDTO();
462         dto.setPk(media.getPk().longValue());
463         dto.setCreatedTime(media.getCreatedTime());
464         dto.setUpdatedTime(media.getUpdatedTime());
465         dto.setMediaUsage(media.getMediaUsage());
466         dto.setMediaStatus(media.getMediaStatus());
467         dto.setMediaStatusInfo(media.getMediaStatusInfo());
468         dto.setFilesetId(media.getFilesetId());
469         dto.setFilesetIuid(media.getFilesetIuid());
470         dto.setMediaCreationRequestIuid(media.getMediaCreationRequestIuid());
471         try {
472             dto.setInstancesAvailable(media.checkInstancesAvailable());
473         } catch (FinderException e) { /* ignore */
474         }
475         return dto;
476     }

538     /**
539      * Returns a dataset for media creation request for given media.
540      * <p>
541      * <DL>
542      * <DT>Set following Tags in dataset.</DT>
543      * <DD>SpecificCharacterSet</DD>
544      * <DD>StorageMediaFileSetID</DD>
545      * <DD>StorageMediaFileSetUID</DD>
546      * <DD>RefSOPSeq with instances of the media.</DD>
547      * </DL>
548      * 
549      * @param pk
550      *            Primary key of the media.
551      * 
552      * @return Prepared Dataset for media creation request.
553      * 
554      * @ejb.interface-method
555      */
556     public Dataset prepareMediaCreationRequest(long pk) throws FinderException {
557         MediaLocal media = mediaHome.findByPrimaryKey(new Long(pk));
558         Dataset ds = DcmObjectFactory.getInstance().newDataset();
559         ds.putCS(Tags.SpecificCharacterSet, "ISO_IR 100");
560         ds.putSH(Tags.StorageMediaFileSetID, media.getFilesetId());
561         ds.putUI(Tags.StorageMediaFileSetUID, media.getFilesetIuid());
562         Collection c = media.getInstances();
563         InstanceLocal il;
564         DcmElement refSOPSeq = ds.putSQ(Tags.RefSOPSeq);
565         for (Iterator iter = c.iterator(); iter.hasNext();) {
566             il = (InstanceLocal) iter.next();
567             Dataset item = refSOPSeq.addNewItem();
568             item.putUI(Tags.RefSOPInstanceUID, il.getSopIuid());
569             item.putUI(Tags.RefSOPClassUID, il.getSopCuid());
570         }
571         return ds;
572     }

574     private void updateSeriesAndStudies(MediaLocal media)
575             throws FinderException {
576         Collection series = seriesHome.findSeriesOnMedia(media);
577         Iterator iter = series.iterator();
578         while (iter.hasNext()) {
579             final SeriesLocal ser = ((SeriesLocal) iter.next());
580             ser.updateFilesetId();
581         }
582         Collection studies = studyHome.findStudiesOnMedia(media);
583         iter = studies.iterator();
584         while (iter.hasNext()) {
585             final StudyLocal sty = ((StudyLocal) iter.next());
586             sty.updateFilesetId();
587         }
588     }

590     /**
591      * Deletes a media.
592      * <p>
593      * Update derived fields from series and studies after media is successfully
594      * deleted.
595      * 
596      * @param mediaPk
597      *            Primary key of the media.
598      * 
599      * @ejb.interface-method
600      */
601     public void deleteMedia(Long mediaPk) throws EJBException, RemoveException,
602             FinderException {
603         MediaLocal media = mediaHome.findByPrimaryKey(mediaPk);
604         Collection series = seriesHome.findSeriesOnMedia(media);
605         Collection studies = studyHome.findStudiesOnMedia(media);
606         String filesetId = media.getFilesetId();
607         mediaHome.remove(mediaPk);
608         log.info("Media " + filesetId + " removed!");
609         Iterator iter = series.iterator();
610         while (iter.hasNext()) {
611             SeriesLocal ser = (SeriesLocal) iter.next();
612             ser.updateFilesetId();
613             ser.updateAvailability();
614         }
615         if (log.isDebugEnabled())
616             log.debug("Series updated after media " + filesetId
617                     + " was deleted!");
618         iter = studies.iterator();
619         while (iter.hasNext()) {
620             StudyLocal sty = (StudyLocal) iter.next();
621             sty.updateFilesetId();
622             sty.updateAvailability();
623         }
624         if (log.isDebugEnabled())
625             log.debug("Studies updated after media " + filesetId
626                     + " was deleted!");
627 
628     }
