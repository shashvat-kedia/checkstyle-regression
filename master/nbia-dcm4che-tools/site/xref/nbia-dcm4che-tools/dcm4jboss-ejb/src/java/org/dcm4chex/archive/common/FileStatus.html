12  import java.util.Arrays;
13  
14  /**
15   * @author gunter.zeilinger@tiani.com

19  public class FileStatus {
20  
21      private static final String[] ENUM = {
22          "MIGRATION_FAILED",
23      	"QUERY_HSM_FAILED",
24      	"MD5_CHECK_FAILED",
25      	"VERIFY_COMPRESS_FAILED", 
26      	"COMPRESS_FAILED",
27      	"DEFAULT", 
28      	"TO_ARCHIVE",
29      	"ARCHIVED",
30          "COMPRESSING",
31          "MIGRATED"
32      };
33  
34      public static final int MIGRATION_FAILED = -5;
35      public static final int QUERY_HSM_FAILED = -4;
36      public static final int MD5_CHECK_FAILED = -3;
37      public static final int VERIFY_COMPRESS_FAILED = -2;
38      public static final int COMPRESS_FAILED = -1;
39      public static final int DEFAULT = 0;    
40      public static final int TO_ARCHIVE = 1;    
41      public static final int ARCHIVED = 2;
42      public static final int COMPRESSING = 3;
43      public static final int MIGRATED = 4;
44      
45      private static final int OFFSET = -MIGRATION_FAILED;
46  
47      public static final String toString(int value) {
48          return ENUM[OFFSET + value];
49      }
50  
51      public static final int toInt(String s) {        
52          final int index = Arrays.asList(ENUM).indexOf(s);
53          if (index == -1)
54              throw new IllegalArgumentException(s);
55          return index - OFFSET;
56      }
57  }
