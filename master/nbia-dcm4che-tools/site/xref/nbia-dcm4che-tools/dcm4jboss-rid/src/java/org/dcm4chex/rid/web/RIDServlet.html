96      /**
97       * Handles the GET requset.
98       * 
99       * @param request 	The http request.
100      * @param response	The http response.
101      * @throws IOException
102      */
103     @Override
104     public void doGet( HttpServletRequest request, HttpServletResponse response ) throws IOException{
105         log.debug("RID URL:"+request.getRequestURI()+"?"+request.getQueryString());
106         if ( request.getParameter("requestType") == null ) {
107             handleMissingRequestType(request, response);
108             return;
109         }
110         BasicRequestObject reqObj = RequestObjectFactory.getRequestObject( request );
111         delegate.getLogger().info("doGet: reqObj:"+reqObj);
112         int reqTypeCode = RIDRequestObject.INVALID_RID_URL;
113         if ( reqObj != null ) {
114             reqTypeCode = reqObj.checkRequest();
115         }
116         if ( reqTypeCode == RIDRequestObject.INVALID_RID_URL) {
117             sendError( response, HttpServletResponse.SC_BAD_REQUEST, "Not a IHE RID URL!" );//missing or wrong request parameter
118             return;
119         }
120         if ( reqTypeCode == RIDRequestObject.RID_REQUEST_NOT_SUPPORTED ) {
121             sendError( response, HttpServletResponse.SC_NOT_FOUND, "This IHE RID request is not supported!" );
122             return;
123         } 
124         RIDResponseObject resp = null; //use RIDResponseObject for encapsulate response.
125         if ( reqTypeCode == RIDRequestObject.SUMMERY_INFO ) {
126             resp = delegate.getRIDSummary( (RIDRequestObject) reqObj );
127         } else if ( reqTypeCode == RIDRequestObject.DOCUMENT ) {
128             resp = delegate.getRIDDocument( (RIDRequestObject) reqObj );
129         } else {
130             sendError( response, HttpServletResponse.SC_NOT_IMPLEMENTED, "This IHE RID request is not yet implemented!" );
131             return;
132         }
133         if ( resp != null ) {
134             int returnCode = resp.getReturnCode();
135             delegate.getLogger().info("doGet: resp returnCode:"+returnCode);
136             if ( returnCode == HttpServletResponse.SC_OK ) {
137                 sendResponse( response, resp );
138             } else if ( returnCode == HttpServletResponse.SC_TEMPORARY_REDIRECT ) {
139                 response.sendRedirect( resp.getErrorMessage() ); //error message contains redirect host.
140             } else {
141                 sendError( response, returnCode, resp.getErrorMessage() );
142             }
143         } else {
144             sendError( response, HttpServletResponse.SC_NOT_FOUND, "Not found!" );
145         }
146     }

148     /**
149      * @param request
150      * @throws IOException
151      */
152     private void handleMissingRequestType(HttpServletRequest request, HttpServletResponse response ) throws IOException {
153         if ("true".equals(this.getServletConfig().getInitParameter("allowShortURL") )) {
154             StringBuffer sb = request.getRequestURL();
155             sb.append("?requestType=").append(request.getParameter("RT"));
156             sb.append("&documentUID=").append(request.getParameter("UID"));
157             sb.append("&preferredContentType=").append(getFullContentType(request.getParameter("PCT")));
158             log.info("redirect shortURL to "+sb);
159             response.sendRedirect(sb.toString());
160         }
161     }
162 
163     /**
164      * @param parameter
165      * @return
166      */
167     private String getFullContentType(String parameter) {
168         if ( parameter.indexOf('/') != -1 ) return parameter;
169         if ( "pdf".equals(parameter) ) return "application/pdf";
170         if ( "xml".equals(parameter) ) return "text/xml";
171         if ( "dcm".equals(parameter) ) return "application/dicom";
172         return parameter;
173     }

175     /**
176      * Send an error response with given response code and message to the client.
177      * <p>
178      * It is recommended that this method is only called once per erequest!<br>
179      * Otherwise a IllegalStateException will be thrown!
180      * 
181      * @param response	The HttpServletResponse of the request.
182      * @param errCode	One of the response code defined in HttpServletResponse.
183      * @param msg		A description for the reason to send the error.
184      */
185     private void sendError( HttpServletResponse response, int errCode, String msg ) {
186         try {
187             response.sendError( errCode, msg );
188         } catch (IOException e) {
189             //delegate.getLogger().error("Cant perform sendError( "+errCode+", "+msg+" )! reason:"+e.getMessage(), e );
190         }
191     }

193     /**
194      * Send the retrieved file to the client.
195      * <p>
196      * Sets the content type as defined in the RIDResponseObject object.
197      * 
198      * @param response
199      * @param respObject
200      */
201     private void sendResponse( HttpServletResponse response, RIDResponseObject respObject ) {
202         log.info("--- sendResponse started! :"+respObject);
203         response.setHeader("Expires","0");//disables client side caching!!!
204         try {
205             if ( respObject != null ) {
206                 response.setContentType( respObject.getContentType() );
207                 long len = respObject.length();
208                 if ( len != -1 ) 
209                     response.setContentLength((int)len);
210                 try {
211                     delegate.getLogger().info("respObject execute");
212                     //respObject.execute( System.out );
213                     respObject.execute( response.getOutputStream() );
214                     response.getOutputStream().close();
215                 } catch ( Exception e ) {
216                     delegate.getLogger().error("Exception while writing RID response to client! reason:"+e.getMessage(), e );
217                 }
218 
219             }
220         } catch ( Exception x ) {
221             delegate.getLogger().error("Exception when handling RID response, will send error response.", x);
222             sendError(response, HttpServletResponse.SC_INTERNAL_SERVER_ERROR, x.getMessage() );
223         }
224         log.info("--- sendResponse finished!");
225     }
