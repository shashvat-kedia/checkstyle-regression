17  package com.google.common.collect.testing;
18  
19  import java.util.Arrays;
20  
21  /**
22   * Minimal GWT emulation of {@code com.google.common.collect.testing.Platform}.
23   *
24   * <p><strong>This .java file should never be consumed by javac.</strong>
25   *
26   * @author Hayward Chan
27   */
28  final class Platform {
29    // Class.cast is not supported in GWT.
30    static void checkCast(Class<?> clazz, Object obj) {}
31  
32    static <T> T[] clone(T[] array) {
33      return (T[]) Arrays.copyOfRange(array, 0, array.length);
34    }
35  
36    // TODO: Consolidate different copies in one single place.
37    static String format(String template, Object... args) {
38      // start substituting the arguments into the '%s' placeholders
39      StringBuilder builder = new StringBuilder(template.length() + 16 * args.length);
40      int templateStart = 0;
41      int i = 0;
42      while (i < args.length) {
43        int placeholderStart = template.indexOf("%s", templateStart);
44        if (placeholderStart == -1) {
45          break;
46        }
47        builder.append(template.substring(templateStart, placeholderStart));
48        builder.append(args[i++]);
49        templateStart = placeholderStart + 2;
50      }
51      builder.append(template.substring(templateStart));
52  
53      // if we run out of placeholders, append the extra args in square braces
54      if (i < args.length) {
55        builder.append(" [");
56        builder.append(args[i++]);
57        while (i < args.length) {
58          builder.append(", ");
59          builder.append(args[i++]);
60        }
61        builder.append("]");
62      }
63  
64      return builder.toString();
65    }
66  
67    private Platform() {}
68  }
