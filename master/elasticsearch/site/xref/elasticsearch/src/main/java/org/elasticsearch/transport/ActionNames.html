122 /**
123  * This class provides the transport action names given the version of the node we are talking to. It was introduced
124  * to maintain backwards compatibility after changing all of the transport action names to better categorize them.
125  * It takes care of the mapping between new action names and previous action names depending on the version of a node.
126  *
127  * The identified categories are:
128  * - indices:admin: apis that allow to perform administration tasks against indices
129  * - indices:data: apis that are about data
130  * - indices:read: apis that read data
131  * - indices:write: apis that write data
132  * - cluster:admin: cluster apis that allow to perform administration tasks
133  * - cluster:monitor: cluster apis that allow to monitor the system
134  * - internal: internal actions that are used from node to node but not directly exposed to users
135  *
136  * Any transport action belongs to one of the above categories and its name starts with its category, followed by a '/'
137  * and the name of the api itself (e.g. cluster:admin/nodes/restart).
138  * When an api exposes multiple transport handlers, some of which are invoked internally during the execution of the api,
139  * we use the `[n]` suffix to identify node actions and the `[s]` suffix to identify shard actions.
140  */

149     /**
150      * Converts, when needed, the action name of an incoming request to the name known by the current node
151      * based on the sender given version
152      */
153     static String incomingAction(String action, Version version) {
154         if (version.before(Version.V_1_4_0_Beta1)) {
155             String post_1_4_action = post_1_4_action(action);
156             //custom action e.g. registered through plugin are not mapped, fallback to the original one
157             if (post_1_4_action != null) {
158                 return post_1_4_action;
159             }
160         }
161         if (version.onOrAfter(Version.V_1_4_0_Beta1) && version.before(Version.V_1_5_0)) {
162             //when we renamed the transport actions in #7105, shard started and failed were flipped around
163             //this didn't cause any actual problem as it was done in a backwards compatible manner.
164             //That said the action names were wrong and we have to fix it still maintaining backwards compatibility
165             if (action.equals(ShardStateAction.SHARD_FAILED_ACTION_NAME)) {
166                 return ShardStateAction.SHARD_STARTED_ACTION_NAME;
167             }
168             if (action.equals(ShardStateAction.SHARD_STARTED_ACTION_NAME)) {
169                 return ShardStateAction.SHARD_FAILED_ACTION_NAME;
170             }
171         }
172         return action;
173     }
174 
175     /**
176      * Converts, when needed, the action name of an outgoing request to the action name known by the recipient node
177      * based on its given version
178      */
179     static String outgoingAction(String action, Version version) {
180         if (version.before(Version.V_1_4_0_Beta1)) {
181             String pre_1_4_Action = pre_1_4_Action(action);
182             //custom actions e.g. registered through plugins are not mapped, fallback to the original one
183             if (pre_1_4_Action != null) {
184                 return pre_1_4_Action;
185             }
186         }
187         if (version.onOrAfter(Version.V_1_4_0_Beta1) && version.before(Version.V_1_5_0)) {
188             //when we renamed the transport actions in #7105, shard started and failed were flipped around
189             //this didn't cause any actual problem as it was done in a backwards compatible manner.
190             //That said the action names were wrong and we have to fix it still maintaining backwards compatibility
191             if (action.equals(ShardStateAction.SHARD_FAILED_ACTION_NAME)) {
192                 return ShardStateAction.SHARD_STARTED_ACTION_NAME;
193             }
194             if (action.equals(ShardStateAction.SHARD_STARTED_ACTION_NAME)) {
195                 return ShardStateAction.SHARD_FAILED_ACTION_NAME;
196             }
197         }
198         return action;
199     }
200 
201     static String post_1_4_action(String action) {
202         return ACTION_NAMES.inverse().get(action);
203     }
204 
205     static String pre_1_4_Action(String action) {
206         return ACTION_NAMES.get(action);
207     }

390     /**
391      * Adds the mapping for an action that exposes a per node endpoint. The suffix of the node endpoint used to be '/n'
392      * and became '[n]' after renaming.
393      */

395         builder.put(name, pre_14_name);

397     }
