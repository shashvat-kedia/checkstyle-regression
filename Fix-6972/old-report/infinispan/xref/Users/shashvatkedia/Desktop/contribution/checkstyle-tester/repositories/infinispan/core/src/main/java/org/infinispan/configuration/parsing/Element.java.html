87      private final String name;
88  
89      Element(final String name) {
90          this.name = name;
91      }
92  
93      /**
94       * Get the local name of this element.
95       *
96       * @return the local name
97       */
98      public String getLocalName() {
99          return name;
100     }
101 
102     private static final Map<String, Element> MAP;
103 
104     static {
105         final Map<String, Element> map = new HashMap<String, Element>(8);
106         for (Element element : values()) {
107             final String name = element.getLocalName();
108             if (name != null) map.put(name, element);
109         }
110         MAP = map;
111     }
112 
113     public static Element forName(String localName) {
114         final Element element = MAP.get(localName);
115         return element == null ? UNKNOWN : element;
116     }
