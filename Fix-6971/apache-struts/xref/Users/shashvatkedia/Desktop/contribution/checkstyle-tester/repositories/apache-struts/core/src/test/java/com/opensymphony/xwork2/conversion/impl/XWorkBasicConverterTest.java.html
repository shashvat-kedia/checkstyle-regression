103     private void compareDates(Locale locale, Object convertedObject) {
104         Calendar cal = Calendar.getInstance(locale);
105         cal.set(Calendar.YEAR, 2009);
106         cal.set(Calendar.MONTH, Calendar.JANUARY);
107         cal.set(Calendar.DATE, 9);
108 
109         Calendar cal1 = Calendar.getInstance(locale);
110         cal1.setTime((Date) convertedObject);
111 
112         assertEquals(cal.get(Calendar.YEAR), cal1.get(Calendar.YEAR));
113         assertEquals(cal.get(Calendar.MONTH), cal1.get(Calendar.MONTH));
114         assertEquals(cal.get(Calendar.DATE), cal1.get(Calendar.DATE));
115 
116         DateFormat df = DateFormat.getDateInstance(DateFormat.SHORT, locale);
117         assertEquals(df.format(cal.getTime()), df.format(convertedObject));
118     }

255         assertEquals(0, BigDecimal.ZERO.compareTo((BigDecimal) convertedObject));

260         assertEquals(0, BigDecimal.ZERO.compareTo((BigDecimal) convertedObject));
261         
262         convertedObject = basicConverter.convertValue(BigDecimal.valueOf(0), BigDecimal.class);
263         assertEquals(BigDecimal.ZERO, convertedObject);
264         assertTrue(convertedObject instanceof BigDecimal);
265         assertEquals(0, BigDecimal.ZERO.compareTo((BigDecimal) convertedObject));
266         
267         convertedObject = basicConverter.convertValue(BigDecimal.valueOf(0.0), BigDecimal.class);
268         assertEquals(BigDecimal.valueOf(0.0), convertedObject);
269         assertTrue(convertedObject instanceof BigDecimal);
270         assertEquals(0, BigDecimal.ZERO.compareTo((BigDecimal) convertedObject));
271 
272         convertedObject = basicConverter.convertValue(BigDecimal.valueOf(0.000), BigDecimal.class);
273         assertEquals(BigDecimal.valueOf(0.000), convertedObject);
274         assertTrue(convertedObject instanceof BigDecimal);
275         assertEquals(0, BigDecimal.ZERO.compareTo((BigDecimal) convertedObject));
276 
277         convertedObject = basicConverter.convertValue(BigDecimal.valueOf(10), BigDecimal.class);
278         assertEquals(BigDecimal.TEN, convertedObject);
279 
280         convertedObject = basicConverter.convertValue(BigDecimal.valueOf(12345.67890), BigDecimal.class);
