249         public String setOption(String key, String value) {
250             String opts;
251             if (value.equals("On")) {
252                 opts = "random";
253             } else if (value.equals("Off")) {
254                 opts = "single";
255             } else if (value.equals("Both")) {
256                 opts = "random,single";
257             } else {
258                 return "Bad value";
259             }
260             return ((Option.ObjectList)paintList).setValueFromString(opts);
261         }

278     public void initContext(TestEnvironment env, GraphicsTests.Context ctx) {
279         super.initContext(env, ctx);
280         RenderTests.Context rctx = (RenderTests.Context) ctx;
281         boolean alphacolor;
282 
283         if (hasGraphics2D) {
284             Graphics2D g2d = (Graphics2D) rctx.graphics;
285             if (env.isEnabled(doAntialias)) {
286                 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
287                                      RenderingHints.VALUE_ANTIALIAS_ON);
288             }
289             alphacolor = env.isEnabled(doAlphaColors);
290             g2d.setStroke((Stroke) env.getModifier(strokeList));
291         } else {
292             alphacolor = false;
293         }
294 
295         String paint = (String)env.getModifier(paintList);
296         if (paint.equals("single")) {
297             Color c = Color.darkGray;
298             if (alphacolor) {
299                 c = makeAlphaColor(c, 32);
300             }
301             rctx.graphics.setColor(c);
302         } else if (paint.equals("random")) {
303             rctx.colorlist = alphacolor ? randAlphaColors : randOpaqueColors;
304         } else if (paint.equals("gradient2")) {
305             Color[] colors = makeGradientColors(2, alphacolor);
306             Graphics2D g2d = (Graphics2D)rctx.graphics;
307             g2d.setPaint(new GradientPaint(0.0f, 0.0f, colors[0],
308                                            10.0f, 10.0f, colors[1], true));
309         } else if (paint.equals("linear2")) {
310             Graphics2D g2d = (Graphics2D)rctx.graphics;
311             g2d.setPaint(makeLinear(2, alphacolor));
312         } else if (paint.equals("linear3")) {
313             Graphics2D g2d = (Graphics2D)rctx.graphics;
314             g2d.setPaint(makeLinear(3, alphacolor));
315         } else if (paint.equals("radial2")) {
316             Graphics2D g2d = (Graphics2D)rctx.graphics;
317             g2d.setPaint(makeRadial(2, alphacolor));
318         } else if (paint.equals("radial3")) {
319             Graphics2D g2d = (Graphics2D)rctx.graphics;
320             g2d.setPaint(makeRadial(3, alphacolor));
321         } else if (paint.equals("texture20")) {
322             Graphics2D g2d = (Graphics2D)rctx.graphics;
323             g2d.setPaint(makeTexturePaint(20, alphacolor));
324         } else if (paint.equals("texture32")) {
325             Graphics2D g2d = (Graphics2D)rctx.graphics;
326             g2d.setPaint(makeTexturePaint(32, alphacolor));
327         } else {
328             throw new InternalError("Invalid paint mode");
329         }
330     }

332     private Color[] makeGradientColors(int numColors, boolean alpha) {
333         Color[] colors = new Color[] {Color.red, Color.blue,
334                                       Color.green, Color.yellow};
335         Color[] ret = new Color[numColors];
336         for (int i = 0; i < numColors; i++) {
337             ret[i] = alpha ? makeAlphaColor(colors[i], 32) : colors[i];
338         }
339         return ret;
340     }
341 
342     private LinearGradientPaint makeLinear(int numColors, boolean alpha) {
343         float interval = 1.0f / (numColors - 1);
344         float[] fractions = new float[numColors];
345         for (int i = 0; i < fractions.length; i++) {
346             fractions[i] = i * interval;
347         }
348         Color[] colors = makeGradientColors(numColors, alpha);
349         return new LinearGradientPaint(0.0f, 0.0f,
350                                        10.0f, 10.0f,
351                                        fractions, colors,
352                                        CycleMethod.REFLECT);
353     }
354 
355     private RadialGradientPaint makeRadial(int numColors, boolean alpha) {
356         float interval = 1.0f / (numColors - 1);
357         float[] fractions = new float[numColors];
358         for (int i = 0; i < fractions.length; i++) {
359             fractions[i] = i * interval;
360         }
361         Color[] colors = makeGradientColors(numColors, alpha);
362         return new RadialGradientPaint(0.0f, 0.0f, 10.0f,
363                                        fractions, colors,
364                                        CycleMethod.REFLECT);
365     }

367     private TexturePaint makeTexturePaint(int size, boolean alpha) {
368         int s2 = size / 2;
369         int type =
370             alpha ? BufferedImage.TYPE_INT_ARGB : BufferedImage.TYPE_INT_RGB;
371         BufferedImage img = new BufferedImage(size, size, type);
372         Color[] colors = makeGradientColors(4, alpha);
373         Graphics2D g2d = img.createGraphics();
374         g2d.setComposite(AlphaComposite.Src);
375         g2d.setColor(colors[0]);
376         g2d.fillRect(0, 0, s2, s2);
377         g2d.setColor(colors[1]);
378         g2d.fillRect(s2, 0, s2, s2);
379         g2d.setColor(colors[3]);
380         g2d.fillRect(0, s2, s2, s2);
381         g2d.setColor(colors[2]);
382         g2d.fillRect(s2, s2, s2, s2);
383         g2d.dispose();
384         Rectangle2D bounds = new Rectangle2D.Float(0, 0, size, size);
385         return new TexturePaint(img, bounds);
386     }

397         public void runTest(Object ctx, int numReps) {
398             RenderTests.Context rctx = (RenderTests.Context) ctx;
399             int size = rctx.size - 1;
400             int x = rctx.initX;
401             int y = rctx.initY;
402             Graphics g = rctx.graphics;
403             g.translate(rctx.orgX, rctx.orgY);
404             Color rCArray[] = rctx.colorlist;
405             int ci = rctx.colorindex;
406             if (rctx.animate) {
407                 do {
408                     if (rCArray != null) {
409                         g.setColor(rCArray[ci++ & NUM_RANDOMCOLORMASK]);
410                     }
411                     g.drawLine(x, y, x + size, y + size);
412                     if ((x -= 3) < 0) x += rctx.maxX;
413                     if ((y -= 1) < 0) y += rctx.maxY;
414                 } while (--numReps > 0);
415             } else {
416                 do {
417                     if (rCArray != null) {
418                         g.setColor(rCArray[ci++ & NUM_RANDOMCOLORMASK]);
419                     }
420                     g.drawLine(x, y, x + size, y + size);
421                 } while (--numReps > 0);
422             }
423             rctx.colorindex = ci;
424             g.translate(-rctx.orgX, -rctx.orgY);
425         }
426     }

649         public int pixelsTouched(GraphicsTests.Context ctx) {
650             /*
651              * Approximation: We figured that the vertical chord connecting
652              * the +45 deg and -45 deg points on the ellipse is about
653              * height/sqrt(2) pixels long.  Likewise, the horizontal chord
654              * connecting the +45 deg and +135 deg points on the ellipse is
655              * about width/sqrt(2) pixels long.  Each of these chords has
656              * a parallel on the opposite side of the respective axis (there
657              * are two horizontal chords and two vertical chords).  Altogether
658              * this gives a reasonable approximation of the total number of
659              * pixels touched by the ellipse, so we have:
660              *     2*(w/sqrt(2)) + 2*(h/sqrt(2))
661              *  == (2/sqrt(2))*(w+h)
662              *  == (sqrt(2))*(w+h)
663              */
664             return (int)(Math.sqrt(2.0)*(ctx.outdim.width+ctx.outdim.height));
665         }

703         public int pixelsTouched(GraphicsTests.Context ctx) {
704             /*
705              * The polygon is a hexagon inscribed inside the square but
706              * missing a triangle at each of the four corners of size
707              * (w/4) by (h/2).
708              *
709              * Putting 2 of these triangles together gives a rectangle
710              * of size (w/4) by (h/2).
711              *
712              * Putting 2 of these rectangles together gives a total
713              * missing rectangle size of (w/2) by (h/2).
714              *
715              * Thus, exactly one quarter of the whole square is not
716              * touched by the filled polygon.
717              */
718             int size = ctx.outdim.width * ctx.outdim.height;
719             return size - (size / 4);
720         }

759         public int pixelsTouched(GraphicsTests.Context ctx) {
760             /*
761              * The two horizontal segments have exactly two pixels per column.
762              * Since the diagonals are more vertical than horizontal, using
763              * h*2 would be a good way to count the pixels in those sections.
764              * We then have to figure out the size of the remainder of the
765              * horizontal lines at top and bottom to get the answer:
766              *
767              *     (diagonals less endpoints)*2 + (horizontals)*2
768              *
769              *  or:
770              *
771              *     (h-2)*2 + ((x+w-1-w/4)-(x+w/4)+1)*2
772              *
773              *  since (w == h == size), we then have:
774              *
775              *     (size - size/4 - 1) * 4
776              */
777             int size = ctx.size;
778             if (size <= 1) {
779                 return 1;
780             } else {
781                 return (size - (size / 4) - 1) * 4;
782             }
783         }

827         public int pixelsTouched(GraphicsTests.Context ctx) {
828             /*
829              * The cubic only touches 2 quadrants in the square, thus
830              * at least half of the square is unfilled.  The integrals
831              * to figure out the exact area are not trivial so for the
832              * other 2 quadrants, I'm going to guess that the cubic only
833              * encloses somewhere between 1/2 and 3/4ths of the pixels

835              * 5/16ths of the total square is filled.
836              */
837             // Note: 2x2 ends up hitting exactly 1 pixel...
838             int size = ctx.size;
839             if (size < 2) size = 2;
840             return size * size * 5 / 16;
841         }
