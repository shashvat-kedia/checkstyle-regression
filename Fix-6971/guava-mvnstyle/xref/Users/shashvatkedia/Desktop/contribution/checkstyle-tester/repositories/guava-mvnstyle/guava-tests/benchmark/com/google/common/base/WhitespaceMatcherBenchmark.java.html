17  package com.google.common.base;
18  
19  import com.google.caliper.BeforeExperiment;
20  import com.google.caliper.Benchmark;
21  import com.google.caliper.Param;
22  import com.google.caliper.runner.CaliperMain;
23  import java.util.BitSet;
24  import java.util.Random;
25  
26  /** Benchmark for the {@link CharMatcher#whitespace} implementation. */
27  public class WhitespaceMatcherBenchmark {
28    private static final int STRING_LENGTH = 10000;
29  
30    private static final String OLD_WHITESPACE_TABLE =
31        "\u0001\u0000\u00a0\u0000\u0000\u0000\u0000\u0000"
32            + "\u0000\u0009\n\u000b\u000c\r\u0000\u0000\u2028\u2029\u0000\u0000\u0000\u0000\u0000"
33            + "\u202f\u0000\u0000\u0000\u0000\u0000\u0000\u0000\u0000\u0020\u0000\u0000\u0000\u0000"
34            + "\u0000\u0000\u0000\u0000\u0000\u0000\u3000\u0000\u0000\u0000\u0000\u0000\u0000\u0000"
35            + "\u0000\u0000\u0000\u0085\u2000\u2001\u2002\u2003\u2004\u2005\u2006\u2007\u2008\u2009"
36            + "\u200a\u0000\u0000\u0000\u0000\u0000\u205f\u1680\u0000\u0000\u180e\u0000\u0000\u0000";
37  
38    public static final CharMatcher OLD_WHITESPACE =
39        new CharMatcher() {
40          @Override
41          public boolean matches(char c) {
42            return OLD_WHITESPACE_TABLE.charAt(c % 79) == c;
43          }
44        };
45  
46    @Param private boolean useNew;
47  
48    @Param({"20", "50", "80"})
49    private int percentMatching;
50  
51    private String teststring;
52    private CharMatcher matcher;
53  
54    public static void main(String[] args) throws Exception {
55      CaliperMain.main(WhitespaceMatcherBenchmark.class, new String[] {});
56    }
57  
58    @BeforeExperiment
59    protected void setUp() {
60      BitSet bitSet = new BitSet();
61      for (int i = 0; i < OLD_WHITESPACE_TABLE.length(); i++) {
62        bitSet.set(OLD_WHITESPACE_TABLE.charAt(i));
63      }
64      bitSet.clear(0);
65      bitSet.clear(1);
66      matcher = useNew ? CharMatcher.whitespace() : OLD_WHITESPACE;
67      teststring = newTestString(new Random(1), bitSet, percentMatching);
68    }
69  
70    @Benchmark
71    public int countIn(int reps) {
72      int result = 0;
73      CharMatcher matcher = this.matcher;
74      String teststring = this.teststring;
75      for (int i = 0; i < reps; i++) {
76        result += matcher.countIn(teststring);
77      }
78      return result;
79    }
80  
81    @Benchmark
82    public int collapseFrom(int reps) {
83      int result = 0;
84      CharMatcher matcher = this.matcher;
85      String teststring = this.teststring;
86      for (int i = 0; i < reps; i++) {
87        result += System.identityHashCode(matcher.collapseFrom(teststring, ' '));
88      }
89      return result;
90    }
91  
92    private static String allMatchingChars(BitSet bitSet) {
93      final char[] result = new char[bitSet.cardinality()];
94      for (int j = 0, c = bitSet.nextSetBit(0); j < result.length; ++j) {
95        result[j] = (char) c;
96        c = bitSet.nextSetBit(c + 1);
97      }
98      return new String(result);
99    }
100 
101   private static String newTestString(Random random, BitSet bitSet, int percentMatching) {
102     final String allMatchingChars = allMatchingChars(bitSet);
103     final char[] result = new char[STRING_LENGTH];
104     // Fill with matching chars.
105     for (int i = 0; i < result.length; i++) {
106       result[i] = allMatchingChars.charAt(random.nextInt(allMatchingChars.length()));
107     }
108     // Replace some of chars by non-matching.
109     int remaining = (int) ((100 - percentMatching) * result.length / 100.0 + 0.5);
110     while (remaining > 0) {
111       final char c = (char) random.nextInt();
112       if (bitSet.get(c)) {
113         final int pos = random.nextInt(result.length);
114         if (bitSet.get(result[pos])) {
115           result[pos] = c;
116           remaining--;
117         }
118       }
119     }
120     return new String(result);
121   }
122 }
