100     protected Dataset doNCreate(ActiveAssociation assoc, Dimse rq,
101             Command rspCmd) throws IOException, DcmServiceException {
102         Association as = assoc.getAssociation();
103         String callingAET = as.getCallingAET();
104         String calledAET = as.getCalledAET();
105         final Command cmd = rq.getCommand();
106         final Dataset mpps = rq.getDataset();
107         final String cuid = cmd.getAffectedSOPClassUID();
108         String iuid = cmd.getAffectedSOPInstanceUID();
109         if (iuid == null) {
110             iuid = rspCmd.getAffectedSOPInstanceUID();
111         }
112         log.debug("Creating MPPS:\n");
113         log.debug(mpps);
114         service.logDIMSE(as, CREATE_XML, mpps);
115         Dataset coerce = service.getCoercionAttributesFor(callingAET,
116                 CREATE_XSL, mpps, as);
117         if (coerce != null) {
118             service.coerceAttributes(mpps, coerce);
119         }
120         checkCreateAttributs(mpps);
121         if (!callingAET.equals(calledAET)) {
122             service.ignorePatientIDForUnscheduled(mpps,
123                     Tags.ScheduledStepAttributesSeq, callingAET);
124             service.supplementIssuerOfPatientID(mpps, as, callingAET, false);
125             service.generatePatientID(mpps,
126                     mpps.getItem(Tags.ScheduledStepAttributesSeq), calledAET);
127             DcmElement ssasq = mpps.get(Tags.ScheduledStepAttributesSeq);
128             for (int i = 0, n = ssasq.countItems(); i < n; i++)
129                 service.supplementIssuerOfAccessionNumber(
130                         ssasq.getItem(i), as, callingAET, false);
131         }
132         mpps.putUI(Tags.SOPClassUID, cuid);
133         mpps.putUI(Tags.SOPInstanceUID, iuid);
134         createMPPS(mpps);
135         mpps.setPrivateCreatorID(PrivateTags.CreatorID);
136         mpps.putAE(PrivateTags.CallingAET, callingAET);
137         mpps.setPrivateCreatorID(null);
138         service.sendMPPSNotification(mpps, MPPSScpService.EVENT_TYPE_MPPS_RECEIVED);
139         return null;
140     }

142     private MPPSManager getMPPSManager()
143             throws HomeFactoryException, RemoteException, CreateException {
144         return ((MPPSManagerHome) EJBHomeFactory.getFactory().lookup(
145                 MPPSManagerHome.class, MPPSManagerHome.JNDI_NAME)).create();
146     }

149     private void createMPPS(Dataset mpps)
150             throws DcmServiceException {
151         try {
152             getMPPSManager().createMPPS(mpps, service.patientMatching());
153         } catch (DcmServiceException e) {
154             throw e;
155         } catch (Exception e) {
156             throw new DcmServiceException(Status.ProcessingFailure, e);
157         }
158     }

161     protected Dataset doNSet(ActiveAssociation assoc, Dimse rq, Command rspCmd)
162             throws IOException, DcmServiceException {
163         Association as = assoc.getAssociation();
164         final String callingAET = as.getCallingAET();
165         final Command cmd = rq.getCommand();
166         final Dataset mpps = rq.getDataset();
167         final String iuid = cmd.getRequestedSOPInstanceUID();
168         log.debug("Set MPPS:\n");
169         log.debug(mpps);
170         service.logDIMSE(as, SET_XML, mpps);
171         Dataset coerce = service.getCoercionAttributesFor(callingAET, SET_XSL,
172                 mpps, as);
173         if (coerce != null) {
174             service.coerceAttributes(mpps, coerce);
175         }
176         checkSetAttributs(mpps);
177         mpps.putUI(Tags.SOPInstanceUID, iuid);
178         updateMPPS(mpps);
179         mpps.setPrivateCreatorID(PrivateTags.CreatorID);
180         mpps.putAE(PrivateTags.CallingAET, callingAET);
181         mpps.setPrivateCreatorID(null);
182         service.sendMPPSNotification(mpps, MPPSScpService.EVENT_TYPE_MPPS_RECEIVED);
183         return null;
184     }

197 	private void checkCreateAttributs(Dataset mpps) throws DcmServiceException {
198         for (int i = 0; i < TYPE1_NCREATE_ATTR.length; ++i) {
199             if (!mpps.containsValue(TYPE1_NCREATE_ATTR[i]))
200                     throw new DcmServiceException(Status.MissingAttributeValue,
201                             "Missing Type 1 Attribute "
202                                     + Tags.toString(TYPE1_NCREATE_ATTR[i]));
203         }
204         DcmElement ssaSq = mpps.get(Tags.ScheduledStepAttributesSeq);
205         for (int i = 0, n = ssaSq.countItems(); i < n; ++i) {
206             if (!ssaSq.getItem(i).containsValue(Tags.StudyInstanceUID))
207                     throw new DcmServiceException(Status.MissingAttributeValue,
208                             "Missing Study Instance UID in Scheduled Step Attributes Seq.");
209         }
210         checkCommonAttributs(mpps);
211         final String status = mpps.getString(Tags.PPSStatus);
212         try {
213             if (PPSStatus.toInt(status) == PPSStatus.IN_PROGRESS) {
214                 return;
215             }
216         } catch (IllegalArgumentException e) {}
217         throw new DcmServiceException(Status.InvalidAttributeValue,
218                 "Invalid MPPS Status: " + status);
219     }
220 
221     private void checkSetAttributs(Dataset mpps) throws DcmServiceException {
222         for (int i = 0; i < ONLY_NCREATE_ATTR.length; ++i) {
223             if (mpps.contains(ONLY_NCREATE_ATTR[i]))
224                     throw new DcmServiceException(Status.ProcessingFailure,
225                             "Cannot update attribute "
226                                     + Tags.toString(ONLY_NCREATE_ATTR[i]));
227         }
228         checkCommonAttributs(mpps);
229         final String status = mpps.getString(Tags.PPSStatus);
230         try {
231             if (status == null || PPSStatus.toInt(status) == PPSStatus.IN_PROGRESS) {
232                 return;
233             }
234         } catch (IllegalArgumentException e) {
235                 throw new DcmServiceException(Status.InvalidAttributeValue,
236                         "Invalid MPPS Status: " + status);
237         }
238         for (int i = 0; i < TYPE1_FINAL_ATTR.length; ++i) {
239             if (!mpps.containsValue(TYPE1_FINAL_ATTR[i]))
240                     throw new DcmServiceException(Status.MissingAttributeValue,
241                             "Missing Type 1 Attribute "
242                                     + Tags.toString(TYPE1_FINAL_ATTR[i]));
243         }
244     }
245 
246     private void checkCommonAttributs(Dataset mpps) throws DcmServiceException {
247         checkCode(mpps, Tags.ProcedureCodeSeq, true);
248         checkCode(mpps, Tags.ReasonForPerformedProcedureCodeSeq, false);
249         checkCode(mpps, Tags.PPSDiscontinuationReasonCodeSeq, true);
250     }
251 
252     private void checkCode(Dataset mpps, int tag, boolean single)
253             throws DcmServiceException {
254         DcmElement codesq = mpps.get(tag);
255         if (codesq == null)
256             return;
257 
258         int n = codesq.countItems();
259         if (n == 0)
260             return;
261 
262         if (n > 1 && single)
263             throw new DcmServiceException(Status.InvalidAttributeValue,
264                     Tags.toString(tag) + " contains " + n + " items");
265 
266         for (int i = 0; i < n; i++) {
267             Dataset code = codesq.getItem(i);
268             if (!code.containsValue(Tags.CodeValue))
269                 throw new DcmServiceException(Status.MissingAttributeValue,
270                         "Missing Code Value in item of " + Tags.toString(tag));
271 
272             if (!code.containsValue(Tags.CodingSchemeDesignator))
273                 throw new DcmServiceException(Status.MissingAttributeValue,
274                         "Missing Coding Scheme Designator in item of "
275                         + Tags.toString(tag));
276         }
277     }
