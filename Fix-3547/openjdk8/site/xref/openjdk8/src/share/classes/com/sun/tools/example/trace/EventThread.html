44  /**
45   * This class processes incoming JDI events and displays them
46   *
47   * @author Robert Field
48   */
49  public class EventThread extends Thread {
50  
51      private final VirtualMachine vm;   // Running VM
52      private final String[] excludes;   // Packages to exclude
53      private final PrintWriter writer;  // Where output goes
54  
55      static String nextBaseIndent = ""; // Starting indent for next thread
56  
57      private boolean connected = true;  // Connected to VM
58      private boolean vmDied = true;     // VMDeath occurred
59  
60      // Maps ThreadReference to ThreadTrace instances
61      private Map<ThreadReference, ThreadTrace> traceMap =
62         new HashMap<>();
63  
64      EventThread(VirtualMachine vm, String[] excludes, PrintWriter writer) {
65          super("event-handler");
66          this.vm = vm;
67          this.excludes = excludes;
68          this.writer = writer;
69      }
70  
71      /**
72       * Run the event handling thread.
73       * As long as we are connected, get event sets off
74       * the queue and dispatch the events within them.
75       */
76      @Override
77      public void run() {
78          EventQueue queue = vm.eventQueue();
79          while (connected) {
80              try {
81                  EventSet eventSet = queue.remove();
82                  EventIterator it = eventSet.eventIterator();
83                  while (it.hasNext()) {
84                      handleEvent(it.nextEvent());
85                  }
86                  eventSet.resume();
87              } catch (InterruptedException exc) {
88                  // Ignore
89              } catch (VMDisconnectedException discExc) {
90                  handleDisconnectedException();
91                  break;
92              }
93          }
94      }
95  
96      /**
97       * Create the desired event requests, and enable
98       * them so that we will get events.
99       * @param excludes     Class patterns for which we don't want events
100      * @param watchFields  Do we want to watch assignments to fields
101      */
102     void setEventRequests(boolean watchFields) {
103         EventRequestManager mgr = vm.eventRequestManager();
104 
105         // want all exceptions
106         ExceptionRequest excReq = mgr.createExceptionRequest(null,
107                                                              true, true);
108         // suspend so we can step
109         excReq.setSuspendPolicy(EventRequest.SUSPEND_ALL);
110         excReq.enable();
111 
112         MethodEntryRequest menr = mgr.createMethodEntryRequest();
113         for (int i=0; i<excludes.length; ++i) {
114             menr.addClassExclusionFilter(excludes[i]);
115         }
116         menr.setSuspendPolicy(EventRequest.SUSPEND_NONE);
117         menr.enable();
118 
119         MethodExitRequest mexr = mgr.createMethodExitRequest();
120         for (int i=0; i<excludes.length; ++i) {
121             mexr.addClassExclusionFilter(excludes[i]);
122         }
123         mexr.setSuspendPolicy(EventRequest.SUSPEND_NONE);
124         mexr.enable();
125 
126         ThreadDeathRequest tdr = mgr.createThreadDeathRequest();
127         // Make sure we sync on thread death
128         tdr.setSuspendPolicy(EventRequest.SUSPEND_ALL);
129         tdr.enable();
130 
131         if (watchFields) {
132             ClassPrepareRequest cpr = mgr.createClassPrepareRequest();
133             for (int i=0; i<excludes.length; ++i) {
134                 cpr.addClassExclusionFilter(excludes[i]);
135             }
136             cpr.setSuspendPolicy(EventRequest.SUSPEND_ALL);
137             cpr.enable();
138         }
139     }

141     /**
142      * This class keeps context on events in one thread.
143      * In this implementation, context is the indentation prefix.
144      */
145     class ThreadTrace {
146         final ThreadReference thread;
147         final String baseIndent;
148         static final String threadDelta = "                     ";
149         StringBuffer indent;
150 
151         ThreadTrace(ThreadReference thread) {
152             this.thread = thread;
153             this.baseIndent = nextBaseIndent;
154             indent = new StringBuffer(baseIndent);
155             nextBaseIndent += threadDelta;
156             println("====== " + thread.name() + " ======");
157         }
158 
159         private void println(String str) {
160             writer.print(indent);
161             writer.println(str);
162         }
163 
164         void methodEntryEvent(MethodEntryEvent event)  {
165             println(event.method().name() + "  --  "
166                     + event.method().declaringType().name());
167             indent.append("| ");
168         }
169 
170         void methodExitEvent(MethodExitEvent event)  {
171             indent.setLength(indent.length()-2);
172         }
173 
174         void fieldWatchEvent(ModificationWatchpointEvent event)  {
175             Field field = event.field();
176             Value value = event.valueToBe();
177             println("    " + field.name() + " = " + value);
178         }
179 
180         void exceptionEvent(ExceptionEvent event) {
181             println("Exception: " + event.exception() +
182                     " catch: " + event.catchLocation());
183 
184             // Step to the catch
185             EventRequestManager mgr = vm.eventRequestManager();
186             StepRequest req = mgr.createStepRequest(thread,
187                                                     StepRequest.STEP_MIN,
188                                                     StepRequest.STEP_INTO);
189             req.addCountFilter(1);  // next step only
190             req.setSuspendPolicy(EventRequest.SUSPEND_ALL);
191             req.enable();
192         }
193 
194         // Step to exception catch
195         void stepEvent(StepEvent event)  {
196             // Adjust call depth
197             int cnt = 0;
198             indent = new StringBuffer(baseIndent);
199             try {
200                 cnt = thread.frameCount();
201             } catch (IncompatibleThreadStateException exc) {
202             }
203             while (cnt-- > 0) {
204                 indent.append("| ");
205             }
206 
207             EventRequestManager mgr = vm.eventRequestManager();
208             mgr.deleteEventRequest(event.request());
209         }
210 
211         void threadDeathEvent(ThreadDeathEvent event)  {
212             indent = new StringBuffer(baseIndent);
213             println("====== " + thread.name() + " end ======");
214         }
215     }
216 
217     /**
218      * Returns the ThreadTrace instance for the specified thread,
219      * creating one if needed.
220      */
221     ThreadTrace threadTrace(ThreadReference thread) {
222         ThreadTrace trace = traceMap.get(thread);
223         if (trace == null) {
224             trace = new ThreadTrace(thread);
225             traceMap.put(thread, trace);
226         }
227         return trace;
228     }
229 
230     /**
231      * Dispatch incoming events
232      */
233     private void handleEvent(Event event) {
234         if (event instanceof ExceptionEvent) {
235             exceptionEvent((ExceptionEvent)event);
236         } else if (event instanceof ModificationWatchpointEvent) {
237             fieldWatchEvent((ModificationWatchpointEvent)event);
238         } else if (event instanceof MethodEntryEvent) {
239             methodEntryEvent((MethodEntryEvent)event);
240         } else if (event instanceof MethodExitEvent) {
241             methodExitEvent((MethodExitEvent)event);
242         } else if (event instanceof StepEvent) {
243             stepEvent((StepEvent)event);
244         } else if (event instanceof ThreadDeathEvent) {
245             threadDeathEvent((ThreadDeathEvent)event);
246         } else if (event instanceof ClassPrepareEvent) {
247             classPrepareEvent((ClassPrepareEvent)event);
248         } else if (event instanceof VMStartEvent) {
249             vmStartEvent((VMStartEvent)event);
250         } else if (event instanceof VMDeathEvent) {
251             vmDeathEvent((VMDeathEvent)event);
252         } else if (event instanceof VMDisconnectEvent) {
253             vmDisconnectEvent((VMDisconnectEvent)event);
254         } else {
255             throw new Error("Unexpected event type");
256         }
257     }
258 
259     /***
260      * A VMDisconnectedException has happened while dealing with
261      * another event. We need to flush the event queue, dealing only
262      * with exit events (VMDeath, VMDisconnect) so that we terminate
263      * correctly.
264      */
265     synchronized void handleDisconnectedException() {
266         EventQueue queue = vm.eventQueue();
267         while (connected) {
268             try {
269                 EventSet eventSet = queue.remove();
270                 EventIterator iter = eventSet.eventIterator();
271                 while (iter.hasNext()) {
272                     Event event = iter.nextEvent();
273                     if (event instanceof VMDeathEvent) {
274                         vmDeathEvent((VMDeathEvent)event);
275                     } else if (event instanceof VMDisconnectEvent) {
276                         vmDisconnectEvent((VMDisconnectEvent)event);
277                     }
278                 }
279                 eventSet.resume(); // Resume the VM
280             } catch (InterruptedException exc) {
281                 // ignore
282             }
283         }
284     }

286     private void vmStartEvent(VMStartEvent event)  {
287          writer.println("-- VM Started --");
288     }

290     // Forward event for thread specific processing
291     private void methodEntryEvent(MethodEntryEvent event)  {
292          threadTrace(event.thread()).methodEntryEvent(event);
293     }
294 
295     // Forward event for thread specific processing
296     private void methodExitEvent(MethodExitEvent event)  {
297          threadTrace(event.thread()).methodExitEvent(event);
298     }
299 
300     // Forward event for thread specific processing
301     private void stepEvent(StepEvent event)  {
302          threadTrace(event.thread()).stepEvent(event);
303     }
304 
305     // Forward event for thread specific processing
306     private void fieldWatchEvent(ModificationWatchpointEvent event)  {
307          threadTrace(event.thread()).fieldWatchEvent(event);
308     }
309 
310     void threadDeathEvent(ThreadDeathEvent event)  {
311         ThreadTrace trace = traceMap.get(event.thread());
312         if (trace != null) {  // only want threads we care about
313             trace.threadDeathEvent(event);   // Forward event
314         }
315     }
316 
317     /**
318      * A new class has been loaded.
319      * Set watchpoints on each of its fields
320      */
321     private void classPrepareEvent(ClassPrepareEvent event)  {
322         EventRequestManager mgr = vm.eventRequestManager();
323         List<Field> fields = event.referenceType().visibleFields();
324         for (Field field : fields) {
325             ModificationWatchpointRequest req =
326                      mgr.createModificationWatchpointRequest(field);
327             for (int i=0; i<excludes.length; ++i) {
328                 req.addClassExclusionFilter(excludes[i]);
329             }
330             req.setSuspendPolicy(EventRequest.SUSPEND_NONE);
331             req.enable();
332         }
333     }
334 
335     private void exceptionEvent(ExceptionEvent event) {
336         ThreadTrace trace = traceMap.get(event.thread());
337         if (trace != null) {  // only want threads we care about
338             trace.exceptionEvent(event);      // Forward event
339         }
340     }
341 
342     public void vmDeathEvent(VMDeathEvent event) {
343         vmDied = true;
344         writer.println("-- The application exited --");
345     }
346 
347     public void vmDisconnectEvent(VMDisconnectEvent event) {
348         connected = false;
349         if (!vmDied) {
350             writer.println("-- The application has been disconnected --");
351         }
352     }
