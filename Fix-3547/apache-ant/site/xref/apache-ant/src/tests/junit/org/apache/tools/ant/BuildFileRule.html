28  /**
29   * Provides access for JUnit tests to execute Ant targets and access execution details (i.e logs).
30   *
31   * Example usage:
32   * <code>
33   * public class MyTest {
34   *
35   *     \@Rule
36   *     public BuildFileRule rule = new BuildFileRule();
37   *
38   *     \@Before
39   *     public void setUp() {
40   *         rule.configureProject("my/and/file.xml");
41   *     }
42   *
43   *     \@Test
44   *     public void testSuccess() {
45   *         rule.executeTarget("passingTaget");
46   *         assertEquals("Incorrect log message", "[taskName] Action Complete", rule.getLog());
47   *     }
48   *
49   *     \@Test
50   *     public void testException() {
51   *         try {
52   *             rule.executeTarget("failingTarget");
53   *             fail("Target should have thrown a BuildException");
54   *         } catch (BuildException ex) {
55   *             assertContains("Exception did not contain correct text", "Could not find compiler on classpath", ex.getMessage());
56   *         }
57   *     }
58   *
59   * }
60   * </code>
61   */
62  public class BuildFileRule extends ExternalResource {
63  
64  	private Project project;
65  
66      private StringBuffer logBuffer;
67      private StringBuffer fullLogBuffer;
68      private StringBuffer outputBuffer;
69      private StringBuffer errorBuffer;
70  
71     
72  
73      /**
74       * Tidies up following a test execution. If the currently configured
75       * project has a <tt>tearDown</tt> target then this will automatically
76       * be called, otherwise this method will not perform any actions.
77       */
78      @Override
79      protected void after()  {
80          if (project == null) {
81              // configureProject has not been called - nothing we can clean-up
82              return;
83          }
84          final String tearDown = "tearDown";
85          if (project.getTargets().containsKey(tearDown)) {
86              project.executeTarget(tearDown);
87          }
88      }
89  
90      /**
91       * Gets the INFO, WARNING and ERROR message from the current execution,
92       * unless the logging level is set above any of these level in which case
93       * the message is excluded.
94       * This is only valid if configureProject() has been called.
95       *
96       * @return The INFO, WARN and ERROR messages in the log.
97       */
98      public String getLog() {
99          return logBuffer.toString();
100     }
101 
102     /**
103      * Gets any messages that have been logged during the current execution, unless
104      * the logging level has been set above the log level defined in the message.
105      *
106      * Only valid if configureProject() has been called.
107      * @return the content of the log.
108      */
109     public String getFullLog() {
110         return fullLogBuffer.toString();
111     }
112 
113     /**
114      * Provides all output sent to the System.out stream during the current execution.
115      * @return all output messages in a single string, normalised to have platform independent line breaks.
116      */
117     public String getOutput() {
118         return cleanBuffer(outputBuffer);
119     }
120 
121     /**
122      * Provides all output sent to the System.err stream during the current execution.
123      * @return all error messages in a single string, normalised to have platform independent line breaks.
124      */
125     public String getError() {
126         return cleanBuffer(errorBuffer);
127     }
128 
129     private String cleanBuffer(StringBuffer buffer) {
130         StringBuilder cleanedBuffer = new StringBuilder();
131         for (int i = 0; i < buffer.length(); i++) {
132             char ch = buffer.charAt(i);
133             if (ch != '\r') {
134                 cleanedBuffer.append(ch);
135             }
136         }
137         return cleanedBuffer.toString();
138     }
139 
140     /**
141      * Sets up to run the named project
142      *
143      * @param  filename name of project file to run
144      */
145     public void configureProject(String filename) throws BuildException {
146         configureProject(filename, Project.MSG_DEBUG);
147     }
148 
149     /**
150      * Sets up to run the named project
151      *
152      * @param  filename name of project file to run
153      */
154     public void configureProject(String filename, int logLevel) throws BuildException {
155         logBuffer = new StringBuffer();
156         fullLogBuffer = new StringBuffer();
157         project = new Project();
158         project.init();
159         File antFile = new File(System.getProperty("root"), filename);
160         project.setProperty("ant.processid", ProcessUtil.getProcessId("<Process>"));
161         project.setProperty("ant.threadname", Thread.currentThread().getName());
162         project.setUserProperty("ant.file" , antFile.getAbsolutePath());
163         project.addBuildListener(new AntTestListener(logLevel));
164         ProjectHelper.configureProject(project, antFile);
165     }
166 
167     /**
168      * Executes a target in the configured Ant build file. Requires #configureProject()
169      * to have been invoked before this call.
170      *
171      * @param  targetName the target in the currently configured build file to run.
172      */
173     public  void executeTarget(String targetName) {
174         outputBuffer = new StringBuffer();
175         PrintStream out = new PrintStream(new AntOutputStream(outputBuffer));
176         errorBuffer = new StringBuffer();
177         PrintStream err = new PrintStream(new AntOutputStream(errorBuffer));
178         logBuffer = new StringBuffer();
179         fullLogBuffer = new StringBuffer();
180 
181         /* we synchronize to protect our custom output streams from being overridden
182          * by other tests executing targets concurrently. Ultimately this would only
183          * happen if we ran a multi-threaded test executing multiple targets at once, and
184          * this protection doesn't prevent a target from internally modifying the output
185          * stream during a test - but at least this scenario is fairly deterministic so
186          * easier to troubleshoot.
187          */
188         synchronized (System.out) {
189             PrintStream sysOut = System.out;
190             PrintStream sysErr = System.err;
191             sysOut.flush();
192             sysErr.flush();
193             try {
194                 System.setOut(out);
195                 System.setErr(err);
196                 project.executeTarget(targetName);
197             } finally {
198                 System.setOut(sysOut);
199                 System.setErr(sysErr);
200             }
201         }
202     }
203 
204     /**
205      * Get the project which has been configured for a test.
206      *
207      * @return the Project instance for this test.
208      */
209     public Project getProject() {
210         return project;
211     }

214     /**
215      * An output stream which saves contents to our buffer.
216      */
217     protected static class AntOutputStream extends OutputStream {
218         private StringBuffer buffer;
219 
220         public AntOutputStream( StringBuffer buffer ) {
221             this.buffer = buffer;
222         }
223 
224         public void write(int b) {
225             buffer.append((char)b);
226         }
227     }

229     /**
230      * Our own personal build listener.
231      */
232     private class AntTestListener implements BuildListener {
233         private int logLevel;
234 
235         /**
236          * Constructs a test listener which will ignore log events
237          * above the given level.
238          */
239         public AntTestListener(int logLevel) {
240             this.logLevel = logLevel;
241         }
242 
243         /**
244          * Fired before any targets are started.
245          */
246         public void buildStarted(BuildEvent event) {
247         }
248 
249         /**
250          * Fired after the last target has finished. This event
251          * will still be thrown if an error occurred during the build.
252          *
253          * @see BuildEvent#getException()
254          */
255         public void buildFinished(BuildEvent event) {
256         }
257 
258         /**
259          * Fired when a target is started.
260          *
261          * @see BuildEvent#getTarget()
262          */
263         public void targetStarted(BuildEvent event) {
264         }
265 
266         /**
267          * Fired when a target has finished. This event will
268          * still be thrown if an error occurred during the build.
269          *
270          * @see BuildEvent#getException()
271          */
272         public void targetFinished(BuildEvent event) {
273         }
274 
275         /**
276          * Fired when a task is started.
277          *
278          * @see BuildEvent#getTask()
279          */
280         public void taskStarted(BuildEvent event) {
281         }
282 
283         /**
284          * Fired when a task has finished. This event will still
285          * be throw if an error occurred during the build.
286          *
287          * @see BuildEvent#getException()
288          */
289         public void taskFinished(BuildEvent event) {
290         }
291 
292         /**
293          * Fired whenever a message is logged.
294          *
295          * @see BuildEvent#getMessage()
296          * @see BuildEvent#getPriority()
297          */
298         public void messageLogged(BuildEvent event) {
299             if (event.getPriority() > logLevel) {
300                 // ignore event
301                 return;
302             }
303 
304             if (event.getPriority() == Project.MSG_INFO ||
305                 event.getPriority() == Project.MSG_WARN ||
306                 event.getPriority() == Project.MSG_ERR) {
307                 logBuffer.append(event.getMessage());
308             }
309             fullLogBuffer.append(event.getMessage());
310         }
311     }
312 
313     public File getOutputDir() {
314         return new File(getProject().getProperty("output"));
315     }
