15  package com.google.common.util.concurrent;
16  
17  import com.google.common.annotations.GwtCompatible;
18  import com.google.common.base.Preconditions;
19  import com.google.common.collect.ForwardingObject;
20  import com.google.errorprone.annotations.CanIgnoreReturnValue;
21  import java.util.concurrent.ExecutionException;
22  import java.util.concurrent.Future;
23  import java.util.concurrent.TimeUnit;
24  import java.util.concurrent.TimeoutException;
25  
26  /**
27   * A {@link Future} which forwards all its method calls to another future. Subclasses should
28   * override one or more methods to modify the behavior of the backing future as desired per the <a
29   * href="http://en.wikipedia.org/wiki/Decorator_pattern">decorator pattern</a>.
30   *
31   * <p>Most subclasses can just use {@link SimpleForwardingFuture}.
32   *
33   * @author Sven Mawson
34   * @since 1.0
35   */
36  @CanIgnoreReturnValue // TODO(cpovirk): Consider being more strict.
37  @GwtCompatible
38  public abstract class ForwardingFuture<V> extends ForwardingObject implements Future<V> {
39    /** Constructor for use by subclasses. */
40    protected ForwardingFuture() {}
41  
42    @Override
43    protected abstract Future<? extends V> delegate();
44  
45    @Override
46    public boolean cancel(boolean mayInterruptIfRunning) {
47      return delegate().cancel(mayInterruptIfRunning);
48    }
49  
50    @Override
51    public boolean isCancelled() {
52      return delegate().isCancelled();
53    }
54  
55    @Override
56    public boolean isDone() {
57      return delegate().isDone();
58    }
59  
60    @Override
61    public V get() throws InterruptedException, ExecutionException {
62      return delegate().get();
63    }
64  
65    @Override
66    public V get(long timeout, TimeUnit unit)
67        throws InterruptedException, ExecutionException, TimeoutException {
68      return delegate().get(timeout, unit);
69    }
70  
71    // TODO(cpovirk): Use standard Javadoc form for SimpleForwarding* class and constructor
72    /**
73     * A simplified version of {@link ForwardingFuture} where subclasses can pass in an already
74     * constructed {@link Future} as the delegate.
75     *
76     * @since 9.0
77     */
78    public abstract static class SimpleForwardingFuture<V> extends ForwardingFuture<V> {
79      private final Future<V> delegate;
80  
81      protected SimpleForwardingFuture(Future<V> delegate) {
82        this.delegate = Preconditions.checkNotNull(delegate);
83      }
84  
85      @Override
86      protected final Future<V> delegate() {
87        return delegate;
88      }
89    }
90  }
