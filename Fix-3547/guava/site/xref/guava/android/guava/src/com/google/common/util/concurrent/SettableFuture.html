15  package com.google.common.util.concurrent;
16  
17  import com.google.common.annotations.Beta;
18  import com.google.common.annotations.GwtCompatible;
19  import com.google.errorprone.annotations.CanIgnoreReturnValue;
20  import org.checkerframework.checker.nullness.compatqual.NullableDecl;
21  
22  /**
23   * A {@link ListenableFuture} whose result can be set by a {@link #set(Object)}, {@link
24   * #setException(Throwable)} or {@link #setFuture(ListenableFuture)} call. It can also, like any
25   * other {@code Future}, be {@linkplain #cancel cancelled}.
26   *
27   * <p>{@code SettableFuture} is the recommended {@code ListenableFuture} implementation when your
28   * task cannot be implemented with {@link ListeningExecutorService}, the various {@link Futures}
29   * utility methods, or {@link ListenableFutureTask}. Those APIs have less opportunity for developer
30   * error. If your needs are more complex than {@code SettableFuture} supports, use {@link
31   * AbstractFuture}, which offers an extensible version of the API.
32   *
33   * @author Sven Mawson
34   * @since 9.0 (in 1.0 as {@code ValueFuture})
35   */
36  @GwtCompatible
37  public final class SettableFuture<V> extends AbstractFuture.TrustedFuture<V> {
38    /**
39     * Creates a new {@code SettableFuture} that can be completed or cancelled by a later method call.
40     */
41    public static <V> SettableFuture<V> create() {
42      return new SettableFuture<V>();
43    }
44  
45    @CanIgnoreReturnValue
46    @Override
47    public boolean set(@NullableDecl V value) {
48      return super.set(value);
49    }
50  
51    @CanIgnoreReturnValue
52    @Override
53    public boolean setException(Throwable throwable) {
54      return super.setException(throwable);
55    }
56  
57    @Beta
58    @CanIgnoreReturnValue
59    @Override
60    public boolean setFuture(ListenableFuture<? extends V> future) {
61      return super.setFuture(future);
62    }
63  
64    private SettableFuture() {}
65  }
